00550         template <typename Predicate> xml_node find_node(Predicate pred) const
00551         {
00552             if (!_root) return xml_node();
00553 
00554             xml_node cur = first_child();
00555             
00556             while (cur._root && cur._root != _root)
00557             {
00558                 if (pred(cur)) return cur;
00559 
00560                 if (cur.first_child()) cur = cur.first_child();
00561                 else if (cur.next_sibling()) cur = cur.next_sibling();
00562                 else
00563                 {
00564                     while (!cur.next_sibling() && cur._root != _root) cur = cur.parent();
