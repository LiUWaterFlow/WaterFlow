00152                                  dataHandler->getTerrainScale(),
00153                                  dataHandler->getDataHeight()));
00154     total = scale;
00155     glUniformMatrix4fv(glGetUniformLocation(program, "MTWMatrix"), 1, GL_FALSE, glm::value_ptr(total));
00156 
00157     // precalculate the inverse since it is a very large model.
00158     glm::mat3 inverseNormalMatrixTrans = glm::transpose(glm::inverse(glm::mat3(total)));
00159     glUniformMatrix3fv(glGetUniformLocation(program, "iNormalMatrixTrans"), 1, GL_FALSE, glm::value_ptr(inverseNormalMatrixTrans));
