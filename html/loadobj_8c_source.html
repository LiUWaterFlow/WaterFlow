00080 static void OBJGetToken(int * tokenType)
00081 {
00082     char c;
00083     char s[255];
00084     int i;
00085     
00086     // 1. skip space. Check for #, skip line when found
00087     c = getc(fp);
00088     while (c == 32 || c == 9 || c == '#')
00089         {
00090             while (c == '#')
00091     while (c != 13 && c != 10 && c != EOF)
00092         c = getc(fp); // Skip comment
00093             c = getc(fp);
00094         }
00095     
00096     // Inspect first character. Bracket, number, other?
00097     
00098     if (c == 13 || c == 10)
00099         {
00100             *tokenType = crlfToken;
00101             //      while (c == 13 && c == 10)
00102             //              c = getc(fp);
00103         }
00104     else
00105         if ((c >= '0' && c <= '9') || c == '-' || c == '.') // Numerical value
00106             {
00107     *tokenType = kInt;
00108     i = 0;
00109     while (c != 13 && c != 10 && c != 32 && c != 9 && c != '/' && c != EOF)
00110         {
00111             if (c == '.' || c == 'E')
00112                 *tokenType = kReal;
00113             s[i++] = c;
00114             c = getc(fp);
00115         }
00116     s[i] = 0;
00117     sscanf(s, "%f", &floatValue[0]);
00118     sscanf(s, "%d", &intValue[0]);
00119     // Check for /
00120     if (c == '/') // parse another number
00121         {
00122             c = getc(fp);
00123             i = 0;
00124             while (c != 13 && c != 10 && c != 32 && c != 9 && c != '/' && c != EOF)
00125                 {
00126         s[i++] = c;
00127         c = getc(fp);
00128                 }
00129             s[i] = 0;
00130             
00131             if (i == 0)
00132                 {
00133         floatValue[1] = -1;
00134         intValue[1] = -1;
00135                 }
00136             else
00137                 {
00138         sscanf(s, "%f", &floatValue[1]);
00139         sscanf(s, "%d", &intValue[1]);
00140                 }
00141             *tokenType = tripletToken;
00142         }
00143     if (c == '/') // parse one more number
00144         {
00145             c = getc(fp);
00146             i = 0;
00147             while (c != 13 && c != 10 && c != 32 && c != 9 && c != '/' && c != EOF)
00148                 {
00149         s[i++] = c;
00150         c = getc(fp);
00151                 }
00152             s[i] = 0;
00153 
00154             if (i == 0)
00155                 {
00156         floatValue[2] = -1;
00157         intValue[2] = -1;
00158                 }
00159             else
00160                 {
00161         sscanf(s, "%f", &floatValue[2]);
00162         sscanf(s, "%i", &intValue[2]);
00163                 }
00164             *tokenType = tripletToken;
00165         }
00166         }
00167         else
00168             if (c == EOF)
00169     {
00170         *tokenType = kEOF;
00171     }
00172             else // Other
00173     {
00174         i = 0;
00175         while (c != 13 && c != 10 && c != 32 && c != 9 && c != EOF)
00176             {
00177                 s[i++] = c;
00178                 c = getc(fp);
00179             }
00180         s[i] = 0;
00181         
00182         *tokenType = kUnknown;
00183         // Compare string to symbols
00184         
00185         if (strcmp(s, "v") == 0)
00186             *tokenType = vToken;

00188             *tokenType = vnToken;
00189         if (strcmp(s, "vt") == 0)
00190             *tokenType = vtToken;

00295 {

00298 
00299     // OBS! Unknown number! Can be one single vertex index or a triplet
00300     do
00301     {
00302         OBJGetToken(&tokenType);
00303

00306         case kReal: // Real should not happen
00307         case kInt:

00309             {
00310                 hasPositionIndices = true;

00322         break;
00323     case tripletToken:
00324         // Triplet (out of which some may be missing)
00325 
00326         if (intValue[0] != 0)
00327         {
00328             hasPositionIndices = true;
00329 
00330             if (theMesh->coordIndex != NULL)
00331             {
00332                 if (intValue[0] > 0)
00333                     theMesh->coordIndex[coordCount] = intValue[0]-1;
00334                 else
00335                     theMesh->coordIndex[coordCount] =
00336                         vertCount+intValue[0];
00337             }
00338         }
00339         if (intValue[1] != 0)
00340         {

00346                     theMesh->textureIndex[coordCount] = intValue[1]-1;
00347                 else
00348                     theMesh->textureIndex[coordCount] =

00350             }

00386             theMesh->normalsIndex[coordCount] = -1;

00393 static void ParseOBJ(MeshPtr theMesh)
00394 {
00395     int tokenType;
00396     
00397     tokenType = 0;
00398     while (tokenType != kEOF)
00399     {
00400         OBJGetToken(&tokenType);
00401         switch (tokenType)
00402         {
00403         case vToken:
00404             ReadOneVertex(theMesh);
00405             break;
00406         case vnToken:
00407             ReadOneNormal(theMesh);
00408             break;
00409         case vtToken:
00410             ReadOneTexture(theMesh);
00411             break;
00412         case fToken:
00413             ReadOneFace(theMesh);
00414             break;
00415         case kReal:
00416             // Ignore
00417             break;
00418         case crlfToken:
00419             break;
00420         case kUnknown:
00421             SkipToCRLF();
00422             //while (tokenType != crlfToken && tokenType != kEOF)
00423             //  OBJGetToken(&tokenType);
00424             break;
00425         }
00426     }

00500 void DecomposeToTriangles(struct Mesh *theMesh)
00501 {
00502     int i, vertexCount, triangleCount;
00503     int *newCoords, *newNormalsIndex, *newTextureIndex;
00504     int newIndex = 0; // Index in newCoords
00505     int first = 0;
00506 
00507     // 1. Bygg om hela modellen till trianglar
00508     // 1.1 Calculate how big the list will become
00509     
00510     vertexCount = 0; // Number of vertices in current polygon
00511     triangleCount = 0; // Resulting number of triangles
00512     for (i = 0; i < theMesh->coordCount; i++)
00513     {
00514         if (theMesh->coordIndex[i] == -1)
00515         {
00516         if (vertexCount > 2) triangleCount += vertexCount - 2;
00517             vertexCount = 0;
00518         }
00519         else
00520         {
00521             vertexCount = vertexCount + 1;
00522         }
00523     }
00524     
00525     fprintf(stderr, "Found %d triangles\n", triangleCount);
00526     
00527     newCoords = (int *)malloc(sizeof(int) * triangleCount * 3);
00528     if (theMesh->normalsIndex != NULL)
00529         newNormalsIndex = (int *)malloc(sizeof(int) * triangleCount * 3);
00530     if (theMesh->textureIndex != NULL)
00531         newTextureIndex = (int *)malloc(sizeof(int) * triangleCount * 3);
00532     
00533     // 1.2 Loop through old list and write the new one
00534     // Almost same loop but now it has space to write the result
00535     vertexCount = 0;
00536     for (i = 0; i < theMesh->coordCount; i++)
00537     {
00538         if (theMesh->coordIndex[i] == -1)
00539         {
00540             first = i + 1;
00541             vertexCount = 0;
00542         }
00543         else
00544         {
00545             vertexCount = vertexCount + 1;
00546 
00547             if (vertexCount > 2)
00548             {
00549                 newCoords[newIndex++] = theMesh->coordIndex[first];
00550                 newCoords[newIndex++] = theMesh->coordIndex[i-1];
00551                 newCoords[newIndex++] = theMesh->coordIndex[i];
00552                 
00553                 if (theMesh->normalsIndex != NULL)
00554                 {
00555                     newNormalsIndex[newIndex-3] = theMesh->normalsIndex[first];
00556                     newNormalsIndex[newIndex-2] = theMesh->normalsIndex[i-1];
00557                     newNormalsIndex[newIndex-1] = theMesh->normalsIndex[i];
00558                 }
00559                 
00560                 // Dito for textures
00561                 if (theMesh->textureIndex != NULL)
00562                 {
00563                     newTextureIndex[newIndex-3] = theMesh->textureIndex[first];
00564                     newTextureIndex[newIndex-2] = theMesh->textureIndex[i-1];
00565                     newTextureIndex[newIndex-1] = theMesh->textureIndex[i];
00566                 }
00567             
00568             }
00569         }
00570     }
00571     
00572     free(theMesh->coordIndex);
00573     theMesh->coordIndex = newCoords;
00574     theMesh->coordCount = triangleCount * 3;
00575     if (theMesh->normalsIndex != NULL)
00576     {
00577         free(theMesh->normalsIndex);
00578         theMesh->normalsIndex = newNormalsIndex;
00579     }
00580     if (theMesh->textureIndex != NULL)
00581     {
00582         free(theMesh->textureIndex);
00583         theMesh->textureIndex = newTextureIndex;
00584     }

00588 static void generateNormals(Mesh* mesh)
00589 {
00590     // If model has vertices but no vertexnormals, generate normals
00591     if (mesh->vertices && !mesh->vertexNormals)
00592     {
00593         int face;
00594         int normalIndex;
00595 
00596         mesh->vertexNormals = (GLfloat *)malloc(3 * sizeof(GLfloat) * mesh->vertexCount);
00597         memset(mesh->vertexNormals, 0, 3 * sizeof(GLfloat) * mesh->vertexCount);
00598 
00599         mesh->normalsCount = mesh->vertexCount;
00600 
00601         mesh->normalsIndex = (int *)malloc(sizeof(GLuint) * mesh->coordCount);
00602         memcpy(mesh->normalsIndex, mesh->coordIndex,
00603             sizeof(GLuint) * mesh->coordCount);
00604 
00605         for (face = 0; face * 3 < mesh->coordCount; face++)
00606         {
00607             int i0 = mesh->coordIndex[face * 3 + 0];
00608             int i1 = mesh->coordIndex[face * 3 + 1];
00609             int i2 = mesh->coordIndex[face * 3 + 2];
00610             
00611             GLfloat* vertex0 = &mesh->vertices[i0 * 3];
00612             GLfloat* vertex1 = &mesh->vertices[i1 * 3];
00613             GLfloat* vertex2 = &mesh->vertices[i2 * 3];
00614 
00615             float v0x = vertex1[0] - vertex0[0];
00616             float v0y = vertex1[1] - vertex0[1];
00617             float v0z = vertex1[2] - vertex0[2];
00618 
00619             float v1x = vertex2[0] - vertex0[0];
00620             float v1y = vertex2[1] - vertex0[1];
00621             float v1z = vertex2[2] - vertex0[2];
00622 
00623             float v2x = vertex2[0] - vertex1[0];
00624             float v2y = vertex2[1] - vertex1[1];
00625             float v2z = vertex2[2] - vertex1[2];
00626 
00627             float sqrLen0 = v0x * v0x + v0y * v0y + v0z * v0z;
00628             float sqrLen1 = v1x * v1x + v1y * v1y + v1z * v1z;
00629             float sqrLen2 = v2x * v2x + v2y * v2y + v2z * v2z;
00630 
00631             float len0 = (sqrLen0 >= 1e-6f) ? (float)sqrt(sqrLen0) : 1e-3f;
00632             float len1 = (sqrLen1 >= 1e-6f) ? (float)sqrt(sqrLen1) : 1e-3f;
00633             float len2 = (sqrLen2 >= 1e-6f) ? (float)sqrt(sqrLen2) : 1e-3f;
00634 
00635             float influence0 = (v0x * v1x + v0y * v1y + v0z * v1z) / (len0 * len1);
00636             float influence1 = -(v0x * v2x + v0y * v2y + v0z * v2z) / (len0 * len2);
00637             float influence2 = (v1x * v2x + v1y * v2y + v1z * v2z) / (len1 * len2);
00638 
00639             float angle0 = (influence0 >= 1.f) ? 0.0f : 
00640                 (influence0 <= -1.f) ? PI : (float)acos(influence0);
00641             float angle1 = (influence1 >= 1.f) ? 0.0f : 
00642                 (influence1 <= -1.f) ? PI : (float)acos(influence1);
00643             float angle2 = (influence2 >= 1.f) ? 0.0f : 
00644                 (influence2 <= -1.f) ? PI : (float)acos(influence2);
00645 
00646             float normalX = v1z * v0y - v1y * v0z;
00647             float normalY = v1x * v0z - v1z * v0x;
00648             float normalZ = v1y * v0x - v1x * v0y;
00649 
00650             GLfloat* normal0 = &mesh->vertexNormals[i0 * 3];
00651             GLfloat* normal1 = &mesh->vertexNormals[i1 * 3];
00652             GLfloat* normal2 = &mesh->vertexNormals[i2 * 3];
00653 
00654             normal0[0] += normalX * angle0;
00655             normal0[1] += normalY * angle0;
00656             normal0[2] += normalZ * angle0;
00657 
00658             normal1[0] += normalX * angle1;
00659             normal1[1] += normalY * angle1;
00660             normal1[2] += normalZ * angle1;
00661 
00662             normal2[0] += normalX * angle2;
00663             normal2[1] += normalY * angle2;
00664             normal2[2] += normalZ * angle2;
00665         }
00666 
00667         for (normalIndex = 0; normalIndex < mesh->normalsCount; normalIndex++)
00668         {
00669             GLfloat* normal = &mesh->vertexNormals[normalIndex * 3];
00670             float length = (float)sqrt(normal[0] * normal[0] + normal[1] * normal[1]
00671                             + normal[2] * normal[2]);
00672             float reciprocalLength = 1.f;
00673 
00674             if (length > 0.01f)
00675                 reciprocalLength = 1.f / length;
00676

00685 static Model* generateModel(Mesh* mesh)
00686 {
00687     // Convert from Mesh format (multiple index lists) to Model format
00688     // (one index list) by generating a new set of vertices/indices
00689     // and where new vertices have been created whenever necessary
00690 
00691     typedef struct
00692     {
00693         int positionIndex;
00694         int normalIndex;
00695         int texCoordIndex;
00696         int newIndex;
00697     } IndexTriplet;
00698 
00699     int hashGap = 6;
00700 
00701     int indexHashMapSize = (mesh->vertexCount * hashGap + mesh->coordCount);
00702 
00703     IndexTriplet* indexHashMap = (IndexTriplet*)malloc(sizeof(IndexTriplet)
00704                             * indexHashMapSize);
00705 
00706     int numNewVertices = 0;
00707     int index;
00708 
00709     int maxValue = 0;
00710         
00711     Model* model = (Model*)malloc(sizeof(Model));
00712     memset(model, 0, sizeof(Model));
00713 
00714     model->indexArray = (GLuint *)malloc(sizeof(GLuint) * mesh->coordCount);
00715     model->numIndices = mesh->coordCount;
00716 
00717     memset(indexHashMap, 0xff, sizeof(IndexTriplet) * indexHashMapSize);
00718 
00719     for (index = 0; index < mesh->coordCount; index++)
00720     {
00721         IndexTriplet currentVertex = { -1, -1, -1, -1 };
00722         int insertPos = 0;
00723         if (mesh->coordIndex)
00724             currentVertex.positionIndex = mesh->coordIndex[index];
00725         if (mesh->normalsIndex)
00726             currentVertex.normalIndex = mesh->normalsIndex[index];
00727         if (mesh->textureIndex)
00728             currentVertex.texCoordIndex = mesh->textureIndex[index];
00729 
00730         if (maxValue < currentVertex.texCoordIndex)
00731             maxValue = currentVertex.texCoordIndex;
00732  
00733         if (currentVertex.positionIndex >= 0)
00734             insertPos = currentVertex.positionIndex * hashGap;
00735 
00736         while (1)
00737         {
00738             if (indexHashMap[insertPos].newIndex == -1)
00739                 {
00740                     currentVertex.newIndex = numNewVertices++;
00741                     indexHashMap[insertPos] = currentVertex;
00742                     break;
00743                 }
00744             else if (indexHashMap[insertPos].positionIndex
00745                  == currentVertex.positionIndex
00746                  && indexHashMap[insertPos].normalIndex
00747                  == currentVertex.normalIndex
00748                  && indexHashMap[insertPos].texCoordIndex
00749                  == currentVertex.texCoordIndex)
00750                 {
00751                     currentVertex.newIndex = indexHashMap[insertPos].newIndex;
00752                     break;
00753                 }
00754             else
00755                 insertPos++;
00756         } 
00757 
00758         model->indexArray[index] = currentVertex.newIndex;
00759     }
00760 
00761     if (mesh->vertices)
00762         model->vertexArray = (GLfloat *)malloc(sizeof(GLfloat) * 3 * numNewVertices);
00763     if (mesh->vertexNormals)
00764         model->normalArray = (GLfloat *)malloc(sizeof(GLfloat) * 3 * numNewVertices);
00765     if (mesh->textureCoords)
00766         model->texCoordArray = (GLfloat *)malloc(sizeof(GLfloat) * 2 * numNewVertices);
00767     
00768     model->numVertices = numNewVertices;
00769 
00770     for (index = 0; index < indexHashMapSize; index++)
00771     {
00772         if (indexHashMap[index].newIndex != -1)
00773         {
00774             if (mesh->vertices)
00775                 memcpy(&model->vertexArray[3 * indexHashMap[index].newIndex],
00776                     &mesh->vertices[3 * indexHashMap[index].positionIndex],
00777                     3 * sizeof(GLfloat));
00778 
00779             if (mesh->vertexNormals)
00780                 memcpy(&model->normalArray[3 * indexHashMap[index].newIndex],
00781                     &mesh->vertexNormals[3 * indexHashMap[index].normalIndex],
00782                     3 * sizeof(GLfloat));
00783 
00784             if (mesh->textureCoords)
00785             {
00786                 model->texCoordArray[2 * indexHashMap[index].newIndex + 0]
00787                     = mesh->textureCoords[2 * indexHashMap[index].texCoordIndex + 0];
00788                 model->texCoordArray[2 * indexHashMap[index].newIndex + 1]
00789                     = 1 - mesh->textureCoords[2 * indexHashMap[index].texCoordIndex + 1];
00790             }
00791         }
00792     }
00793 
00794     free(indexHashMap);
00795 
00796     return model;
00797 }
