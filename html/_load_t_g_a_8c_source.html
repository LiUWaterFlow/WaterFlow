00204     glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);   // Linear Filtered

00206     if (texture->bpp == 8)                      // Was The TGA 8 Bits? Should be grayscale then.
00207     {
00208         type=GL_RED;            // If So Set The 'type' To GL_RED
00209     }
00210     if (texture->bpp == 24)                     // Was The TGA 24 Bits?
00211     {
00212         type=GL_RGB;            // If So Set The 'type' To GL_RGB
00213     }
00214     glTexImage2D(GL_TEXTURE_2D, 0, type, texture->w, texture->h, 0, type, GL_UNSIGNED_BYTE, texture[0].imageData);

00218         glGenerateMipmap(GL_TEXTURE_2D);
