00025 char* readFile(const char *file)
00026 {
00027     FILE *fptr;
00028     long length;
00029     char *buf;
00030 
00031     if (file == NULL) {
00032         return NULL;
00033     }
00034 
00035     fptr = fopen(file, "rb"); /* Open file for reading */
00036     if (!fptr) {
00037         fprintf(stderr, "Could not open file: %s!\n", file);
00038         return NULL;/* Return NULL on failure */
00039     }
00040 
00041     int temp = fseek(fptr, 0, SEEK_END); /* Seek to the end of the file */
00042     if (temp != 0) {
00043         fprintf(stderr, "Could not seek to end of file!\n");
00044         return NULL;
00045     }
00046 
00047     length = ftell(fptr); /* Find out how many bytes into the file we are */
00048     if (length < 0) {
00049         fprintf(stderr, "ftell reported negative length!\n");
00050         return NULL;
00051     }
00052 
00053     buf = (char*)malloc(length + 1); /* Allocate a buffer for the entire length of the file and a null terminator */
00054     if (buf == NULL) {
00055         fprintf(stderr, "Could not allocate space for readFile buffer!\n");
00056         return NULL;
00057     }
00058 
00059     fseek(fptr, 0, SEEK_SET); /* Go back to the beginning of the file */
00060     fread(buf, length, 1, fptr); /* Read the contents of the file in to the buffer */
00061     fclose(fptr); /* Close the file */
00062     buf[length] = 0; /* Null terminator */
00063 
00064     return buf; /* Return the buffer */
00065 }

00112 GLuint compileShaders(const char *vs, const char *fs, const char *gs, const char *tcs, const char *tes,
00113                                 const char *vfn, const char *ffn, const char *gfn, const char *tcfn, const char *tefn)
00114 {
00115     GLuint v,f,g,tc,te,p;
00116     v = glCreateShader(GL_VERTEX_SHADER);
00117     f = glCreateShader(GL_FRAGMENT_SHADER);
00118 
00119     glShaderSource(v, 1, &vs, NULL);
00120     glShaderSource(f, 1, &fs, NULL);
00121     glCompileShader(v);
00122     glCompileShader(f);
00123 
00124     if (gs != NULL)
00125     {
00126         g = glCreateShader(GL_GEOMETRY_SHADER);
00127         glShaderSource(g, 1, &gs, NULL);
00128         glCompileShader(g);
00129     }
00130 #ifdef GL_TESS_CONTROL_SHADER
00131     if (tcs != NULL)
00132     {
00133         tc = glCreateShader(GL_TESS_CONTROL_SHADER);
00134         glShaderSource(tc, 1, &tcs, NULL);
00135         glCompileShader(tc);
00136     }
00137     if (tes != NULL)
00138     {
00139         te = glCreateShader(GL_TESS_EVALUATION_SHADER);
00140         glShaderSource(te, 1, &tes, NULL);
00141         glCompileShader(te);
00142     }
00143 #endif
00144     p = glCreateProgram();
00145     glAttachShader(p,v);
00146     glAttachShader(p,f);
00147     if (gs != NULL)
00148         glAttachShader(p,g);
00149     if (tcs != NULL)
00150         glAttachShader(p,tc);
00151     if (tes != NULL)
00152         glAttachShader(p,te);
00153     glLinkProgram(p);
00154     glUseProgram(p);
00155 
00156     printShaderInfoLog(v, vfn);
00157     printShaderInfoLog(f, ffn);
00158     if (gs != NULL)     printShaderInfoLog(g, gfn);
00159     if (tcs != NULL)    printShaderInfoLog(tc, tcfn);
00160     if (tes != NULL)    printShaderInfoLog(te, tefn);
00161 
00162     // We can remove the shaders since the data is now linked to the program
00163     glDeleteShader(v);
00164     glDeleteShader(f);
00165     if (gs != NULL)     glDeleteShader(g);
00166     if (tcs != NULL)    glDeleteShader(tc);
00167     if (tes != NULL)    glDeleteShader(te);
00168 
00169     printProgramInfoLog(p, vfn, ffn, gfn, tcfn, tefn);
00170 
00171     return p;
00172 }

00231 /* report GL errors, if any, to stderr */
00232 void printError(const char *functionName)
00233 {
00234    GLenum error;
00235    while (( error = glGetError() ) != GL_NO_ERROR)
00236    {
00237        if ((lastError != error) || (strcmp(functionName, lastErrorFunction)))
00238        {
00239            fprintf (stderr, "GL error 0x%X detected in %s\n", error, functionName);
00240            strcpy(lastErrorFunction, functionName);
00241            lastError = error;
00242        }
00243    }

00290 // create FBO
00291 // FP buffer, suitable for HDR
00292 FBOstruct *initFBO(int width, int height, int int_method)
00293 {
00294     FBOstruct *fbo = (FBOstruct*)malloc(sizeof(FBOstruct));
00295 
00296     if (fbo == NULL) {
00297         fprintf(stderr, "initFBO could not allocate memory for FBO!\n");
00298         return NULL;
00299     }
00300 
00301     fbo->width = width;
00302     fbo->height = height;
00303 
00304     // create objects
00305     glGenFramebuffers(1, &fbo->fb); // frame buffer id
00306     glBindFramebuffer(GL_FRAMEBUFFER, fbo->fb);
00307     glGenTextures(1, &fbo->texid);
00308     //fprintf(stderr, "%i \n",fbo->texid);
00309     glBindTexture(GL_TEXTURE_2D, fbo->texid);
00310     glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
00311     glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
00312     if (int_method == 0)
00313     {
00314         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
00315         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
00316     }
00317     else
00318     {
00319         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
00320         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
00321     }
00322     glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA32F, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, NULL);
00323     glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, fbo->texid, 0);
00324 
00325     // Renderbuffer
00326     // initialize depth renderbuffer
00327     glGenRenderbuffers(1, &fbo->rb);
00328     glBindRenderbuffer(GL_RENDERBUFFER, fbo->rb);
00329     glRenderbufferStorage( GL_RENDERBUFFER, GL_DEPTH_COMPONENT24, fbo->width, fbo->height );
00330     glFramebufferRenderbuffer( GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, fbo->rb );
00331     CHECK_FRAMEBUFFER_STATUS();
00332 
00333     //fprintf(stderr, "Framebuffer object %d\n", fbo->fb);
00334     glBindFramebuffer(GL_FRAMEBUFFER, 0);
00335     return fbo;

00518 // choose input (textures) and output (FBO)
00519 void useFBO(FBOstruct *out, FBOstruct *in1, FBOstruct *in2)
00520 {
00521     GLint curfbo;
00522 
00523 // This was supposed to catch changes in viewport size and update lastw/lasth.
00524 // It worked for me in the past, but now it causes problems to I have to
00525 // fall back to manual updating.
00526     glGetIntegerv(GL_FRAMEBUFFER_BINDING, &curfbo);
00527     if (curfbo == 0)
00528     {
00529         GLint viewport[4] = {0,0,0,0};
00530         GLint w, h;
00531         glGetIntegerv(GL_VIEWPORT, viewport);
00532         w = viewport[2] - viewport[0];
00533         h = viewport[3] - viewport[1];
00534         if ((w > 0) && (h > 0) && (w < 65536) && (h < 65536)) // I don't believe in 64k pixel wide frame buffers for quite some time
00535         {
00536             lastw = viewport[2] - viewport[0];
00537             lasth = viewport[3] - viewport[1];
00538         }
00539     }
00540 
00541     if (out != 0L)
00542         glViewport(0, 0, out->width, out->height);
00543     else
00544         glViewport(0, 0, lastw, lasth);
00545 
00546     if (out != 0L)
00547     {
00548         glBindFramebuffer(GL_FRAMEBUFFER, out->fb);
00549         glViewport(0, 0, out->width, out->height);
00550     }
00551     else
00552         glBindFramebuffer(GL_FRAMEBUFFER, 0);
00553     glActiveTexture(GL_TEXTURE1);
00554     if (in2 != 0L)
00555         glBindTexture(GL_TEXTURE_2D, in2->texid);
00556     else
00557         glBindTexture(GL_TEXTURE_2D, 0);
