00025 char* readFile(char *file)
00026 {
00027     FILE *fptr;
00028     long length;
00029     char *buf;
00030     if (file == NULL)
00031         return NULL;
00032     fptr = fopen(file, "rb"); /* Open file for reading */
00033     if (!fptr) /* Return NULL on failure */
00034         return NULL;
00035     fseek(fptr, 0, SEEK_END); /* Seek to the end of the file */
00036     length = ftell(fptr); /* Find out how many bytes into the file we are */
00037     buf = (char*)malloc(length+1); /* Allocate a buffer for the entire length of the file and a null terminator */
00038     fseek(fptr, 0, SEEK_SET); /* Go back to the beginning of the file */
00039     fread(buf, length, 1, fptr); /* Read the contents of the file in to the buffer */
00040     fclose(fptr); /* Close the file */
00041     buf[length] = 0; /* Null terminator */
00042 
00043     return buf; /* Return the buffer */
00044 }

00136     if (gs != NULL) printShaderInfoLog(g, gfn);
00137     if (tcs != NULL)    printShaderInfoLog(tc, tcfn);
00138     if (tes != NULL)    printShaderInfoLog(te, tefn);
00139 
00140     printProgramInfoLog(p, vfn, ffn, gfn, tcfn, tefn);

00172     if ((gs==NULL) && (geomFileName != NULL))
00173         fprintf(stderr, "Failed to read %s from disk.\n", geomFileName);
00174     if ((tcs==NULL) && (tcFileName != NULL))

00176     if ((tes==NULL) && (teFileName != NULL))
00177         fprintf(stderr, "Failed to read %s from disk.\n", teFileName);
00178     if ((vs!=NULL)&&(fs!=NULL))
00179         p = compileShaders(vs, fs, gs, tcs, tes, vertFileName, fragFileName, geomFileName, tcFileName, teFileName);
00180     if (vs != NULL) free(vs);
00181     if (fs != NULL) free(fs);
00182     if (gs != NULL) free(gs);
00183     if (tcs != NULL) free(tcs);

00202 /* report GL errors, if any, to stderr */
00203 void printError(const char *functionName)
00204 {
00205    GLenum error;
00206    while (( error = glGetError() ) != GL_NO_ERROR)
00207    {
00208        if ((lastError != error) || (strcmp(functionName, lastErrorFunction)))
00209        {
00210            fprintf (stderr, "GL error 0x%X detected in %s\n", error, functionName);
00211            strcpy(lastErrorFunction, functionName);
00212            lastError = error;
00213        }
00214    }
00215 }
